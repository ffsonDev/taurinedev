// Обработка данных
import "std/json.tau" as json
import "std/array.tau" as array

let data = json.parse('{ "users": [{"name": "John"}, {"name": "Jane"}] }')

for user in data.users {
    print(f"Привет, {user.name}!")
}

// Функциональные методы
let names = array.map(data.users, u => u.name)
print(names)
// Встраивание в Rust
use taurine::Interpreter;

fn main() -> Result<(), String> {
    let mut vm = Interpreter::new();

    // Зарегистрировать Rust функцию
    vm.register_fn("rust_add", |a, b| a + b)?;

    // Выполнить Taurine код
    vm.run(r#"
        let result = rust_add(10, 20)
        print(f"Результат: {result}")
    "#)?;

    Ok(())
}

// Быстрый прототип алгоритма
function quicksort(arr) {
    if #arr <= 1 { return arr }

    let pivot = arr[#arr / 2]
    let left = []
    let right = []

    for i in 0..#arr {
        if i != #arr / 2 {
            if arr[i] < pivot {
                array.push(left, arr[i])
            } else {
                array.push(right, arr[i])
            }
        }
    }

    return quicksort(left) + [pivot] + quicksort(right)
}

Установка

1

Из исходников

git clone https://github.com/ffsonDev/taurine.git
cd taurine
cargo build --release
2

Проверка

taurine --version
# 🐂 Taurine v1.0.0
3

Запуск

taurine script.tau      # Запуск файла
taurine --repl           # Интерактивный режим
taurine --optimize script.tau  # С оптимизацией

💡 Совет: Добавьте target/release в PATH для удобного доступа к команде taurine
