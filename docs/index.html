Руководство по языку Taurine

Taurine — быстрый, встраиваемый скриптовый язык, реализованный на Rust. Он сочетает простоту Lua с современными функциями языка и производительностью компилируемых языков.

Цели дизайна

  • Простой — Лёгкий в изучении и использовании
  • Быстрый — Оптимизирован для однопоточной производительности
  • Встраиваемый — Легко интегрируется в Rust приложения
  • Современный — f-строки, деструктуризация, nil-safe операторы

Начало работы

Установка

# Из исходников
git clone https://github.com/ffsonDev/taurine.git
cd taurine
cargo build --release

# Или через cargo-binstall
cargo install cargo-binstall
cargo binstall taurine

Запуск скриптов

# Запуск скрипта
taurine script.tau

# Запуск REPL (интерактивный режим)
taurine --repl

# С оптимизациями
taurine --optimize script.tau

# Форматирование кода
taurine --format script.tau

Hello World

print("Привет, Taurine!")
print(f"2 + 2 = {2 + 2}")

Базовый синтаксис

Комментарии

// Однострочный комментарий

/* Многострочные комментарии пока не поддерживаются */

Переменные

// Изменяемая переменная
let x = 10
x = 100

// PI = 3  // Ошибка!

// Legacy синтаксис (всё ещё работает)
loc y = 20

Идентификаторы

  • Начинаются с буквы или подчёркивания
  • Могут содержать буквы, цифры, подчёркивания
  • Чувствительны к регистру
let name = "тест"
let _private = "скрытое"
let camelCase = true
let snake_case = true

Типы данных

Числа

64-битные числа с плавающей точкой:

let int = 42
let float = 3.14
let negative = -10
let scientific = 1e10

Строки

UTF-8 строки с escape-последовательностями:

let str = "Привет, Мир!"
let multiline = "Строка1\nСтрока2"
let tabbed = "Кол1\tКол2"
let quoted = "Он сказал \"Привет\""

Булевы значения

let truthy = true
let falsy = false

Nil

Представляет отсутствие значения:

let empty = nil

Массивы

Упорядоченные списки значений:

let numbers = [1, 2, 3, 4, 5]
let mixed = [1, "два", true, nil]
let nested = [[1, 2], [3, 4]]

Таблицы

Ассоциативные массивы (как объекты/словари):

let person = { name: "Иван", age: 30 }
let empty = {}
let nested = { outer: { inner: "значение" } }

Диапазоны

Используются в циклах for-in:

// 1 до 9 (исключая 10)
for i in 1..10 { print(i) }

Операторы

Арифметические

let sum = 1 + 2       // 3
let diff = 5 - 3      // 2
let prod = 4 * 5      // 20
let quot = 10 / 2     // 5
let neg = -5          // -5

Сравнения

let eq = (1 == 1)     // true
let ne = (1 != 2)     // true
let lt = (1 < 2)      // true
let gt = (2 > 1)      // true
let le = (1 <= 1)     // true
let ge = (2 >= 2)     // true

Логические

let and = (true and false)  // false
let or = (true or false)    // true
let not = not true          // false

Конкатенация строк

let str = "Привет" + " " + "Мир"  // "Привет Мир"
let mixed = "Значение: " + 42     // "Значение: 42"

Длина

let len = #[1, 2, 3]    // 3
let slen = #"привет"    // 6
let tlen = #{a: 1, b: 2} // 2

Управляющие конструкции

If/Else

if x > 10 {
    print("большое")
} else if x > 5 {
    print("среднее")
} else {
    print("маленькое")
}

While цикл

let i = 0
while i < 10 {
    print(i)
    i = i + 1
}

For-In цикл

// Диапазон
for i in 1..10 {
    print(i)
}

// Массив
for item in [1, 2, 3] {
    print(item)
}

// Строка (посимвольно)
for ch in "привет" {
    print(ch)
}

Break/Continue

for i in 1..10 {
    if i == 5 {
        break  // Выход из цикла
    }
    if i == 3 {
        continue  // Переход к следующей итерации
    }
    print(i)
}

Функции

Определение

function add(a, b) {
    return a + b
}

Параметры по умолчанию

function greet(name, greeting = "Привет") {
    print(f"{greeting}, {name}!")
}

greet("Мир")           // "Привет, Мир!"
greet("Мир", "Hi")     // "Hi, Мир!"

Multi-Return

Функции могут возвращать несколько значений:

function divmod(a, b) {
    return a / b, a % b
}

// Захват всех возвращаемых значений
let result = divmod(10, 3)

// Деструктуризация
let {q, r} = divmod(10, 3)
print(f"Частное: {q}, Остаток: {r}")

Функциональные литералы

let add = function(a, b) {
    return a + b
}

// В таблицах
let obj = {
    name: "тест",
    greet: function(self) {
        print(f"Привет, {self.name}!")
    }
}

Лямбда-функции

let double = x => x * 2
let sum = (a, b) => a + b

// Использование с массивами
import "std/array.tau" as array
let arr = [1, 2, 3]
let doubled = array.map(arr, x => x * 2)

Таблицы и массивы

Операции с массивами

import "std/array.tau" as array

let arr = [1, 2, 3]

// Длина
let len = #arr

// Доступ
let first = arr[0]
arr[0] = 100

// Операции
array.push(arr, 4)
let last = array.pop(arr)
let doubled = array.map(arr, x => x * 2)
let evens = array.filter(arr, x => x % 2 == 0)

Операции с таблицами

let obj = { name: "тест", value: 42 }

// Доступ
print(obj.name)
print(obj["name"])

// Изменение
obj.name = "новый"
obj.newField = "добавлено"

// Nil-safe доступ
print(obj?.missing)  // nil (без ошибки)

Методы таблиц

let counter = { count: 0 }

function increment(self, amount) {
    self.count = self.count + amount
}

increment(counter, 5)
print(counter.count)  // 5

Обработка ошибок

Try/Catch

try {
    risky_operation()
} catch (err) {
    print(f"Ошибка: {err}")
}

Throw

function divide(a, b) {
    if b == 0 {
        throw "Деление на ноль"
    }
    return a / b
}

Assertions

import "std/test.tau" as test

assert(x > 0, "x должно быть положительным")
assert_eq(a, b, "a и b должны быть равны")

F-строки

Интерполяция строк с помощью f-строк:

let name = "Taurine"
let version = "1.0"

print(f"Привет от {name} v{version}!")
print(f"2 + 2 = {2 + 2}")
print(f"Результат: {10 * 5}")

Multi-Return

Функции могут возвращать несколько значений, которые можно деструктуризировать:

function getPerson() {
    return "Иван", 30, "Москва"
}

// Деструктуризация
let {name, age, city} = getPerson()
print(f"{name}, {age} лет, из {city}")

Nil-safe операторы

Безопасный доступ к свойствам без риска ошибки:

let obj = { name: "тест" }

// Safe property access
print(obj?.name)     // "тест"
print(obj?.missing)  // nil (без ошибки)

let nilObj = nil
print(nilObj?.prop)  // nil

Деструктуризация

Извлечение значений из массивов и таблиц:

// Деструктуризация массива
let arr = [1, 2, 3]
let {first, second} = arr

// Деструктуризация таблицы
let person = { name: "Иван", age: 30, city: "Москва" }
let {name, age} = person
print(f"{name}, {age}")  // "Иван, 30"

// С переименованием
let {name: userName, age: userAge} = person
print(f"{userName}, {userAge}")
