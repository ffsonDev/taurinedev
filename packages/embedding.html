Руководство по встраиванию

Taurine может быть встроен в другие приложения, позволяя использовать его как скриптовый язык в ваших программах.

Введение

Taurine разработан для лёгкой интеграции с другими языками программирования. Вы можете:

  • Выполнять код Taurine из вашего приложения
  • Вызывать функции Taurine из хост-языка
  • Регистрировать функции хост-языка для использования в Taurine
  • Получать и устанавливать переменные Taurine

Базовая архитектура

Rust API

Наиболее полный и производительный способ встраивания.

Установка

Добавьте Taurine в ваш Cargo.toml:

[dependencies]
taurine = "1.0"

Базовое использование

Запуск кода Taurine

use taurine::{Interpreter, Value};

fn main() -> Result<(), String> {
    // Создать интерпретатор
    let mut interp = Interpreter::new();

    // Запустить код на Taurine
    interp.run(r#"
        let x = 10
        let y = 20
        print(f"x + y = {x + y}")
    "#)?;

    Ok(())
}

Вызов функций Taurine из Rust

use taurine::{Interpreter, Value};

fn main() -> Result<(), String> {
    let mut interp = Interpreter::new();

    // Определить функцию на Taurine
    interp.run(r#"
        function add(a, b) {
            return a + b
        }
    "#)?;

    // Вызвать функцию из Rust
    let result = interp.call("add", &[
        Value::Number(5.0), 
        Value::Number(3.0)
    ])?;

    if let Value::Number(n) = result {
        println!("Результат: {}", n); // 8.0
    }

    Ok(())
}

Получение значений из Taurine

use taurine::{Interpreter, Value};

fn main() -> Result<(), String> {
    let mut interp = Interpreter::new();

    // Установить переменную в Taurine
    interp.run(r#"
        let message = "Привет из Rust!"
        let numbers = [1, 2, 3, 4, 5]
    "#)?;

    // Получить переменную
    let message = interp.get("message")?;
    let numbers = interp.get("numbers")?;

    println!("Сообщение: {:?}", message);
    println!("Числа: {:?}", numbers);

    Ok(())
}

Регистрация Rust функций

use taurine::{Interpreter, Value};

fn main() -> Result<(), String> {
    let mut interp = Interpreter::new();

    // Зарегистрировать Rust функцию
    interp.register_fn("rust_add", |args: &[Value]| {
        if args.len() < 2 {
            return Err("Ожидается 2 аргумента".to_string());
        }

        if let (Value::Number(a), Value::Number(b)) = (&args[0], &args[1]) {
            Ok(Value::Number(a + b))
        } else {
            Err("Ожидаются числа".to_string())
        }
    });

    // Использовать функцию в Taurine
    interp.run(r#"
        let result = rust_add(10, 20)
        print(f"Результат Rust функции: {result}")
    "#)?;

    Ok(())
}

Обработка ошибок

use taurine::{Interpreter, Value};

fn main() {
    let mut interp = Interpreter::new();

    match interp.run(r#"
        let x = "hello"
        let y = x + 10  // Это вызовет ошибку
    "#) {
        Ok(_) => println!("Успех!"),
        Err(e) => eprintln!("Ошибка: {}", e),
    }
}

Продвинутые возможности Rust API

Компиляция и повторное выполнение

use taurine::{Interpreter, CompiledCode};

fn main() -> Result<(), String> {
    let mut interp = Interpreter::new();

    // Компилировать код один раз
    let compiled: CompiledCode = interp.compile(r#"
        function compute(x, y) {
            return x * y + 10
        }
    "#)?;

    // Выполнять многократно
    for i in 0..5 {
        interp.run_compiled(&compiled)?;
        let result = interp.call("compute", &[
            Value::Number(i as f64),
            Value::Number(2.0)
        ])?;
        println!("Iteration {}: {:?}", i, result);
    }

    Ok(())
}

C API

Taurine предоставляет C API для встраивания в C/C++ приложения.

Подключение

#include <stdio.h>
#include "taurine.h"

Пример на C

#include <stdio.h>
#include "taurine.h"

int main() {
    // Создать VM
    TaurineVM* vm = taurine_new();

    // Запустить код на Taurine
    const char* code =
        "let x = 10\n"
        "let y = 20\n"
        "print(f\"x + y = {x + y}\")\n";

    if (taurine_run(vm, code) != 0) {
        fprintf(stderr, "Ошибка: %s\n", taurine_get_error(vm));
        taurine_free(vm);
        return 1;
    }

    // Получить значение
    TaurineValue* val = taurine_get(vm, "x");
    if (taurine_is_number(val)) {
        printf("x = %f\n", taurine_as_number(val));
    }

    // Очистить
    taurine_value_free(val);
    taurine_free(vm);

    return 0;
}

C++ API

Объектно-ориентированный wrapper для C API.

Пример на C++

#include <iostream>
#include <string>
#include "taurine.h"

class Taurine {
private:
    TaurineVM* vm;

public:
    Taurine() : vm(taurine_new()) {}
    ~Taurine() { taurine_free(vm); }

    bool run(const std::string& code) {
        return taurine_run(vm, code.c_str()) == 0;
    }

    std::string getError() const {
        return taurine_get_error(vm);
    }

    double getNumber(const std::string& name) {
        TaurineValue* val = taurine_get(vm, name.c_str());
        double result = taurine_as_number(val);
        taurine_value_free(val);
        return result;
    }

    std::string getString(const std::string& name) {
        TaurineValue* val = taurine_get(vm, name.c_str());
        std::string result = taurine_as_string(val);
        taurine_value_free(val);
        return result;
    }
};

int main() {
    Taurine taurine;

    if (!taurine.run(R"(
        let pi = 3.14159
        let radius = 5
        let area = pi * radius * radius
        let message = "Площадь круга"
    )")) {
        std::cerr << "Ошибка: " << taurine.getError() << std::endl;
        return 1;
    }

    double area = taurine.getNumber("area");
    std::string msg = taurine.getString("message");
    
    std::cout << msg << ": " << area << std::endl;

    return 0;
}

Python Bindings

Использование Taurine из Python приложений.

Установка

pip install taurine

Пример использования

import taurine

# Создать интерпретатор
vm = taurine.Interpreter()

# Выполнить код
vm.run("""
    let x = 10
    let y = 20
    print(f"x + y = {x + y}")
""")

# Получить значение
result = vm.get("x")
print(f"Значение x: {result}")

# Вызвать функцию
vm.run("""
    function multiply(a, b) {
        return a * b
    }
""")

result = vm.call("multiply", [5, 3])
print(f"5 * 3 = {result}")

# Зарегистрировать Python функцию
def python_add(a, b):
    return a + b

vm.register_function("py_add", python_add)
vm.run("print(py_add(10, 20))")

Советы по производительности

1. Повторное использование VM

// ❌ Плохо: создание VM для каждого вызова
fn process_script(code: &str) {
    let vm = Interpreter::new();
    vm.run(code);
}

// ✅ Хорошо: переиспользование VM
struct ScriptProcessor {
    vm: Interpreter,
}

impl ScriptProcessor {
    fn new() -> Self {
        Self {
            vm: Interpreter::new(),
        }
    }

    fn process(&mut self, code: &str) {
        self.vm.run(code);
    }
}

2. Кэширование кода

// Компилируйте код один раз, выполняйте многократно
let compiled = vm.compile(script_code)?;

for item in items {
    vm.set("item", item);
    vm.run_compiled(&compiled)?;
}

3. Минимизация FFI вызовов

// ❌ Плохо: много вызовов через FFI
for i in 0..1000 {
    vm.call("process", &[Value::Number(i as f64)])?;
}

// ✅ Хорошо: пакетная обработка
vm.run(r#"
    for i in 0..1000 {
        process(i)
    }
"#)?;

4. Использование нативных функций

// Регистрируйте критичный код как Rust функции
vm.register_fn("fast_compute", |a: f64, b: f64| {
    // Оптимизированная реализация
    (a * b).sqrt() + a.sin()
});

Рекомендации по безопасности

1. Песочница

use taurine::{Interpreter, Sandbox};

let mut vm = Interpreter::new();

// Включить песочницу
let sandbox = Sandbox::new()
    .disable_io()
    .disable_network()
    .disable_filesystem();

vm.set_sandbox(sandbox);

2. Ограничение времени выполнения

use std::time::Duration;

vm.set_timeout(Duration::from_secs(5));

match vm.run(long_running_script) {
    Ok(_) => println!("Успех!"),
    Err(e) if e.contains("timeout") => {
        eprintln!("Превышено время выполнения");
    }
    Err(e) => eprintln!("Ошибка: {}", e),
}

3. Ограничение памяти

vm.set_memory_limit(100 * 1024 * 1024); // 100 MB

match vm.run(memory_intensive_script) {
    Ok(_) => println!("Успех!"),
    Err(e) if e.contains("memory") => {
        eprintln!("Превышен лимит памяти");
    }
    Err(e) => eprintln!("Ошибка: {}", e),
}

4. Фильтрация опасных функций

// Не регистрируйте опасные функции для недоверенного кода
// - io.remove()
// - os.exec()
// - process.spawn()

// Или создайте whitelist разрешённых функций
let allowed_functions = vec!["print", "math.sin", "string.upper"];
vm.set_function_whitelist(allowed_functions);
